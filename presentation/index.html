Why Lodash?

Lodash's modular methods are great for...
  • Iterating arrays, objects, strings
  • Manipulating & testing values
  • Creating composite functions

Getting started

In a browser: 

<script src="lodash.js"></script>

Using npm: 

$ npm i -g npm
$ npm i --save lodash

In Node.js: 

// Load the full build.
var _ = require('lodash');
// Load the core build.
var _ = require('lodash/core');
 
// Load method categories.
var array = require('lodash/array');
var object = require('lodash/fp/object');
 
// Cherry-pick methods for smaller browserify/rollup/webpack bundles.
var at = require('lodash/at');
var curryN = require('lodash/fp/curryN');

Lodash.js features

_.chunk(array, [size=1])


_.chunk(['a', 'b', 'c', 'd'], 2);
// => [['a', 'b'], ['c', 'd']]
 
_.chunk(['a', 'b', 'c', 'd'], 3);
// => [['a', 'b', 'c'], ['d']]

Returns the new array of chunks

_.drop / _.dropRight (array, [n=1])


_.drop([1, 2, 3], 2);
// => [3]

_.dropRight([1, 2, 3], 2);
// => [1]

Returns the slice of array

_.flatten(array)


var array = [1, [2, [3, [4]], 5]];

_.flatten(array);
// => [1, 2, [3, [4]], 5]

_.flattenDeep(array);
// => [1, 2, 3, 4, 5]

_.flattenDepth(array, 2);
// => [1, 2, 3, [4], 5]

Returns the new flattened array

_.uniq(array)


_.uniq([2, 1, 2]);
// => [2, 1]

Returns the new duplicate free array

_.zip([arrays])


_.zip(['a', 'b'], [1, 2], [true, false]);
// => [['a', 1, true], ['b', 2, false]]

Returns the new array of grouped elements

_.debounce(func, [wait=0], [options={}])


function validateEmail() {
  // Validate email here and show error message if not valid
}

let emailInput = document.getElementById("email-field");
emailInput.addEventListener("keyup", _.debounce(validateEmail, 500));

Returns the new debounced function

_.gt / _.gte (value, other)


_.gt(3, 1);
// => true
 
_.gt(3, 3);
// => false

_.gte(3, 1);
// => true
 
_.gte(3, 3);
// => true

Returns true if value is greater (or equal in case _.gte) than other, else false

_.isArrayLike(value)


_.isArrayLike([1, 2, 3]);
// => true
 
_.isArrayLike(document.body.children);
// => true
 
_.isArrayLike('abc');
// => true
 
_.isArrayLike(_.noop);
// => false

Returns true if value is array-like, else false

_.capitalize([string=''])


_.capitalize('FRED');
// => 'Fred'

_.capitalize('uniq');
// => 'Uniq'

_.capitalize('AbRAKadabRa');
// => 'Abrakadabra'

Returns the capitalized string

_.trim([string=''], [chars=whitespace])


_.trim('  abc  ');
// => 'abc'
 
_.trim('-_-abc-_-', '_-');
// => 'abc'

Returns the trimmed string
